#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <unistd.h>

typedef enum { RED, GREEN, YELLOW } TrafficLightState;

void delay(int seconds) {
    time_t start_time = time(NULL);
    while (time(NULL) < start_time + seconds);
}

void changeTrafficLight(TrafficLightState* state) {
    if (*state == RED)
        *state = GREEN;
    else if (*state == GREEN)
        *state = YELLOW;
    else
        *state = RED;
}

int main() {
    TrafficLightState trafficLight = RED;
    for (int i = 0; i < 10; i++) {
        printf("Traffic light is %s\n", trafficLight == RED ? "RED" : (trafficLight == GREEN ? "GREEN" : "YELLOW"));
        delay(2);  // Simulate a 2-second interval
        changeTrafficLight(&trafficLight);
    }
    return 0;
}
